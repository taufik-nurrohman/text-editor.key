Text Editor Key 1.0.0

Key extension for text editor.

Provides a feature to easily interact with the keyboard keys.

Try to press Control-b or Control-i in the text area above!

Usage

Browser

<script src="./text-editor/index.min.js"></script>
<script src="./text-editor.key/index.min.js"></script>
<script>

  const editor = new TextEditor(document.querySelector('textarea'), {
      with: [TextEditor.Key]
  });

  editor.command('italic', function () {
      return this.wrap('<i>', '</i>'), false;
  });

  editor.key('Control-b', function () {
      return this.wrap('<b>', '</b>'), false;
  });

  editor.key('Control-i', 'italic');

</script>

CommonJS

const TextEditor = require('@taufik-nurrohman/text-editor').default;
const Key = require('@taufik-nurrohman/text-editor.key').default;

TextEditor.state.with.push(Key); // Load as a core extension

const editor = new TextEditor(document.querySelector('textarea'), {
    with: [Key] // Load as an optional extension
});

editor.command('italic', function () {
    return this.wrap('<i>', '</i>'), false;
});

editor.key('Control-b', function () {
    return this.wrap('<b>', '</b>'), false;
});

editor.key('Control-i', 'italic');
